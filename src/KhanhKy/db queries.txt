
///// CREATE THE FOOD TABLE
CREATE TABLE IF NOT EXISTS food ( id smallint unsigned UNIQUE NOT NULL auto_increment, 
name varchar(25) NOT NULL, 
viet_name varchar(25) ,  
category varchar (10) NOT NULL,
viet_category varchar (10) ,
price decimal (4,1) NOT NULL,
ingr_name varchar (20) ,
ingr_amount decimal (6,1) ,
primary key (id) );

//// CREATE THE ORDER TABLE
CREATE TABLE IF NOT EXISTS order (
name varchar(9) NOT NULL unique, 
order_id int(6) unique NOT NULL,
status TINYINT(1) NOT NULL,
primary key (name) ),


//// CREATE THE ALL_TABLE TABLE
CREATE TABLE IF NOT EXISTS all_tables (
name varchar(16) NOT NULL unique, 
order_id int(6),
status TINYINT(1) NOT NULL,
primary key (order_id) ),
foreign key (order_id) REFERENCES order(order_id);

//// CREATE THE ITEM TABLE
CREATE TABLE IF NOT EXISTS item (
name varchar(16) NOT NULL unique, 
order_id int(6) unique NOT NULL,
primary key (name) ),
foreign key (order_id) REFERENCES order(order_id);
